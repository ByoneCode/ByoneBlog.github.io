    # 创建配置初始化函数

    //配置初始化
func Configinit(filename string,data interface{}) (err error){
	t := reflect.TypeOf(data)
	//判断data参数是否是指针类型
	if t.Kind() != reflect.Ptr{
		err = fmt.Errorf("data type error!you must use pointers type")
		return
	}
	//打开配置文件
	file, err := ioutil.ReadFile(filename)
	if err != nil {
		err = fmt.Errorf("readfile error!")
		return
	}
	//读取每一行的数据
	lineStr := strings.Split(string(file),"\r\n")
	//取出的结构体名
	var structName string
	//遍历每一行的数据
	for idx,line := range lineStr{
		//去掉每一行的首尾空格
		line = strings.TrimSpace(line)
		//如果是#开头的则省略（注释）
		if strings.HasPrefix(line,"#") || len(line) == 0{
			continue
		}
		//判断配置名是否以[开头
		if strings.HasPrefix(line,"[") {
			if line[0] != '[' || line[len(line)-1] != ']'{
				//返回语法错误
				err = fmt.Errorf("syntax error!line:%d",idx+1)
				return
			}
			//获取配置名
			configName := strings.TrimSpace(line[1:len(line)-1])
			//fmt.Println(configName)
			//根据data数据去找配置名
			v := reflect.ValueOf(data).Elem()
			//遍历data数据的配置名
			for i := 0; i < v.NumField();i++  {
				field := t.Elem().Field(i)
				//判断配置文件的配置名和data数据的配置名是否一样
				if configName == field.Tag.Get("ini"){
					//说明找到了该配置名并记下
					structName = field.Name
					fmt.Println(configName,structName)

				}

			}
		}else { //如果不是以[开头的那就参数值
			//判断是否有等于号没有则返回错误
			if strings.Index(line,"=") == -1 {
				err = fmt.Errorf("syntax error!line:%d",idx+1)
				return
			}

			//取出它的键值
			index := strings.Split(line,"=")
			key := strings.TrimSpace(index[0])
			value := strings.TrimSpace(index[1])
			//fmt.Println(key,value)
			//根据structname把结构体取出来
			v := reflect.ValueOf(data)
			sValue := v.Elem().FieldByName(structName) //拿到结构体的值
			sType := sValue.Type() //拿到结构体类型

			//判断data配置是否是一个结构体
			if sType.Kind() != reflect.Struct {
				err = fmt.Errorf("[%s] must be a struct!",structName)
				return
			}
			var fieldName string  //字段名
			var fieldType reflect.StructField
			//遍历结构体的每一个字段
			for i := 0;i < sValue.NumField() ; i++ {
				field := sType.Field(i)
				fieldType = field
				if field.Tag.Get("ini") == key {
					//找到了对应的字段
					fieldName = field.Name
					break
				}
			}

		fieldObj := sValue.FieldByName(fieldName)
		//fmt.Println(fieldName,fieldType.Type.Kind())
		//对结构体进行赋值
			switch fieldType.Type.Kind() {
			case reflect.String:
				fieldObj.SetString(value)
			case reflect.Int,reflect.Int32,reflect.Int8,reflect.Int16,reflect.Int64:
				// 将切割的字符串类型转为Int64类型
				valueInt,error := strconv.ParseInt(value,10,64)
				if error != nil {
					err = fmt.Errorf("params error!line:%d,type:Int",idx+1)
					return
				}
				fieldObj.SetInt(valueInt)
			case reflect.Bool:
				valueBool,error := strconv.ParseBool(value)
				if error != nil {
					err = fmt.Errorf("params error!line:%d,type:Bool",idx+1)
				}
				fieldObj.SetBool(valueBool)
			}
		}
	}
	return
}

    //入口函数
func main()  {
	var cfg Config
	err := Configinit("env.ini",&cfg)
	if err != nil {
		fmt.Printf("Configinit load error!err:%v",err)
		return
	}
	fmt.Println("Configinit load success!")
	fmt.Println(cfg)
}
