    Python简单爬取图片保存

    # 准备前提

    • Python 3.x

    # 实现代码

    # coding = utf-8
import requests
from bs4 import BeautifulSoup
import re
import os

global_url = "https://www.woyaogexing.com"  # 要爬取的网址
global_i = 1   # 文件名
html = requests.get(global_url+"/touxiang/")  # get访问网页源码
html.encoding = "utf-8" # 定义utf-8格式
html = html.text # text输出
soup = BeautifulSoup(html,"html.parser")  # 解析源码
all_img = soup.select('div.txList a.img') # 通过css类名查找元素

def getHtml(url):
  getHtml = requests.get(global_url+url)
  getHtml.encoding = "utf-8"
  getHtml = getHtml.text
  soup = BeautifulSoup(getHtml,"html.parser")
  img = soup.select('li[class="tx-img"]')
  dir_name = "avatar"
  
  if not os.path.exists(dir_name):
    os.mkdir(dir_name)  # 创建文件夹
  for item in img: 
    global global_i  #函数里要用global定义全局变量
    global_i += 1
    rep = re.compile('href="(.*?)"')
    reg = re.findall(rep,str(item))
    reponse = requests.get('https:'+reg[0])
    with open(dir_name+'/'+str(global_i)+'.jpeg','wb') as f:
      f.write(reponse.content)  # 保存图片
    print('保存第'+str(global_i)+'张成功！')

for item in all_img:
  img_link = item.get('href') # 获取a标签的href
  img_title = item.get('title') # 获取a标签的title
  print('链接：'+img_link+'标题：'+img_title)
  getHtml(img_link)
  print("=================================================")
