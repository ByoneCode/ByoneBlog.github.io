    thinkphp6实现验证码验证

    ByoneJie 2021-9-11 thinkphp

    # 后端处理

    # 利用composer安装库

    composer require topthink/think-captcha
    1

    如果安装的时候发生了报错可以在thinkphp项目的根目录找到composer.json文件在尾部添加下面代码（没有发生报错可以省略此步）

    "repositories": {
	    "packagist": {
	       "type": "composer",
	        "url": "https://packagist.phpcomposer.com"
	    }

    # 新建一个Test.php

    <?php 

namespace app\controller;
//导入验证码库
use think\captcha\facade\Captcha;
class Test(){
        //创建验证码
	public function verify(){
		return Captcha::create();    
	}
  //验证验证码
    public function check(){
      //获取前端传过来的验证码参数
      $value = input('get.val')
      //这里使用内置函数手动验证
      if(!captcha_check($value))
      {
              //验证错误事件....
        return json(['msg' => '验证码错误','status' => 0,'code' => 1001]);
      }
    }
}

    # 前端处理

    配置axios并设置全局配置

    const instance = axios.create({ //创建axios实例
	timeout:10*1000, //超时时间
	withCredentials:true, //允许传入cookie
	baseURL:''  //请求链接
})

    封装get请求

    //封装get请求
export const get = (url,param,config = {}) => {
	return new Promise((resolve,reject) => {
		instance({
			method:'get',
			url,
			param,
			...config
		}).then(res => {
			resolve(res)
		}).catch(err => {
			reject(err)
		})
	})
}

    新建一个Test.vue

    <template>
	<div class="test">
		<input type="text" v-model="val">
		<img :src="verify" @click="reload">
		<button @click="check">验证</button>
	</div>
</template>

<script>	
	export default {
		name:'Test',
		data(){
			return {
				val:null,
				vkey:0
			}
		},
		computed:{
			//监听验证码刷新
			verify(){
				return `后端域名/test/verify?key=${this.vkey}`
			}
		},
		methods:{
			//刷新验证码
			reload(){
				this.vkey = Math.ceil(Math.random()*999999)
			},
			check(){
				get('后端域名/test/check',{val}).then(res => {
					console.log(res)
				})
			}
		}
	}
</script>

    # 注意事项

    切记axios必须开启withCredentials:true不然无法传入cookie并进行验证，配置该属性后后端跨域请求链接必须为准确域名不能使用‘*’号，否则会报错
